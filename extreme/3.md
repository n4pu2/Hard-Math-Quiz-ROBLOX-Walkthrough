\log_{x}{7x-10} &= 2 \\
x^{\log_{x}{7x-10}} &= x^2 \\
7x-10 &= x^2 \\
0 &= x^2 - 7x + 10 \\

0 &= x(x-2) -5(x-2) \\

\end{align}
$$
